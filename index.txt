    <script>
        const FORMATS = ['.jpg', '.png', '.webp', '.jpeg'];
        let cachedExt = null;
        let levelData = [];
        let completedLevels = JSON.parse(localStorage.getItem('puz_comp_v31') || '[]');
        let currentLevelId, rows, cols, pWidth, pHeight, pieces = [];

        async function discoverLevels() {
            // é€»è¾‘ï¼šå·²å®Œæˆå…³å¡ + 3ä¸ªé¢„åŠ è½½å…³å¡
            const maxIdx = completedLevels.length > 0 ? Math.max(...completedLevels) : 0;
            const targetCount = maxIdx + 3;
            
            const newList = [];
            for (let i = 1; i <= targetCount; i++) {
                const url = await findImage(i);
                if (url) newList.push({ id: i, url });
                else break;
            }
            levelData = newList;
            document.getElementById('loading-hint').innerText = `å·²å‘ç° ${levelData.length} ä¸ªå…³å¡`;
            renderMenu();
        }

        async function findImage(id) {
            if (cachedExt) {
                const url = `${id}${cachedExt}`;
                if (await check(url)) return url;
            }
            for (const ext of FORMATS) {
                if (ext === cachedExt) continue;
                const url = `${id}${ext}`;
                if (await check(url)) { cachedExt = ext; return url; }
            }
            return null;
        }

        function check(url) {
            return new Promise(r => {
                const i = new Image();
                i.onload = () => r(true);
                i.onerror = () => r(false);
                i.src = url;
            });
        }

        // ä¿®å¤2ï¼šå·²å®Œæˆå…³å¡ç‚¹å‡»å¼¹å‡ºé¢„è§ˆï¼Œæœªå®Œæˆè¿›å…¥æ¸¸æˆ
        function renderMenu() {
            const grid = document.getElementById('level-grid');
            grid.innerHTML = '';
            levelData.forEach(level => {
                const isDone = completedLevels.includes(level.id);
                const card = document.createElement('div');
                card.className = 'level-card';
                
                // åªæœ‰å·²å®Œæˆæˆ–â€œå½“å‰å…³â€æ‰æ˜¾ç¤ºç¼©ç•¥å›¾
                const canSee = isDone || level.id === (completedLevels.length > 0 ? Math.max(...completedLevels)+1 : 1);
                
                card.innerHTML = `
                    <div class="card-thumb">${canSee ? `<img src="${level.url}">` : 'ğŸ”’'}</div>
                    <div style="padding:10px; text-align:center; font-weight:bold;">ç¬¬ ${level.id} å…³</div>
                `;
                
                // ç‚¹å‡»é€»è¾‘ï¼šå·²å®Œæˆåˆ™é¢„è§ˆï¼Œæœªå®Œæˆåˆ™å¼€ç©
                card.onclick = () => isDone ? showPreview(level) : startLevel(level.id);
                grid.appendChild(card);
            });
        }

        function showPreview(level) {
            document.getElementById('preview-img').src = level.url;
            document.getElementById('preview-title').innerText = `ç¬¬ ${level.id} å…³ Â· å·²é€šå…³`;
            document.getElementById('replay-btn').onclick = () => {
                closePreview();
                startLevel(level.id);
            };
            document.getElementById('preview-overlay').style.display = 'flex';
        }

        function closePreview() {
            document.getElementById('preview-overlay').style.display = 'none';
        }

        function startLevel(id) {
            const level = levelData.find(l => l.id === id);
            if (!level) return;
            currentLevelId = id;
            const img = new Image();
            img.src = level.url;
            img.onload = () => {
                const ratio = img.naturalWidth / img.naturalHeight;
                const D = 3 + Math.floor((id - 1) / 3);
                if (ratio > 1.2) { cols = D + 1; rows = D; }
                else if (ratio < 0.8) { cols = D; rows = D + 1; }
                else { cols = D; rows = D; }

                const maxW = Math.min(window.innerWidth * 0.9, 600);
                const maxH = window.innerHeight * 0.7;
                pWidth = Math.floor(maxW / cols);
                pHeight = Math.floor(pWidth / ratio);
                if (pHeight * rows > maxH) {
                    pHeight = Math.floor(maxH / rows);
                    pWidth = Math.floor(pHeight * ratio);
                }
                
                initBoard(level.url);
                document.getElementById('menu-screen').classList.remove('active');
                document.getElementById('game-screen').classList.add('active');
                document.getElementById('game-title').innerText = `ç¬¬ ${id} å…³`;
                document.getElementById('diff-label').innerText = `${cols}x${rows}`;
            };
        }

        function initBoard(url) {
            pieces = [];
            for (let i = 0; i < rows * cols; i++) {
                pieces.push({ id: i, currentPos: i, bgX: (i % cols) * pWidth, bgY: Math.floor(i / cols) * pHeight, url });
            }
            let pool = [...Array(rows * cols).keys()].sort(() => Math.random() - 0.5);
            pieces.forEach((p, i) => p.currentPos = pool[i]);
            renderPieces();
        }

        function renderPieces() {
            const board = document.getElementById('puzzle-board');
            board.innerHTML = '';
            board.style.width = `${pWidth * cols}px`;
            board.style.height = `${pHeight * rows}px`;
            board.style.gridTemplateColumns = `repeat(${cols}, ${pWidth}px)`;

            pieces.sort((a,b) => a.currentPos - b.currentPos).forEach(p => {
                const div = document.createElement('div');
                div.className = 'piece';
                div.dataset.id = p.id;
                div.style.width = `${pWidth}px`;
                div.style.height = `${pHeight}px`;
                div.style.backgroundImage = `url(${p.url})`;
                div.style.backgroundPosition = `-${p.bgX}px -${p.bgY}px`;
                div.style.backgroundSize = `${pWidth * cols}px ${pHeight * rows}px`;
                div.onpointerdown = e => handleDrag(e, p.id);
                board.appendChild(div);
            });
        }

        // ç®€æ˜“æ‹–æ‹½é€»è¾‘ (å¤ç”¨ä¹‹å‰ç‰ˆæœ¬)
        function handleDrag(e, id) {
            const startX = e.clientX, startY = e.clientY;
            const piece = pieces.find(p => p.id === id);
            const el = document.querySelector(`[data-id="${id}"]`);
            el.style.zIndex = "100";

            window.onpointermove = (ev) => {
                el.style.transform = `translate(${ev.clientX - startX}px, ${ev.clientY - startY}px)`;
            };

            window.onpointerup = (ev) => {
                window.onpointermove = null; window.onpointerup = null;
                const dx = ev.clientX - startX, dy = ev.clientY - startY;
                const dc = Math.round(dx / pWidth), dr = Math.round(dy / pHeight);
                
                if (dc !== 0 || dr !== 0) {
                    const nr = Math.floor(piece.currentPos / cols) + dr;
                    const nc = (piece.currentPos % cols) + dc;
                    if (nr >= 0 && nr < rows && nc >= 0 && nc < cols) {
                        const targetPos = nr * cols + nc;
                        const other = pieces.find(p => p.currentPos === targetPos);
                        other.currentPos = piece.currentPos;
                        piece.currentPos = targetPos;
                    }
                }
                el.style.zIndex = "1"; el.style.transform = "";
                renderPieces();
                checkWin();
            };
        }

        function checkWin() {
            if (pieces.every(p => p.id === p.currentPos)) {
                if (!completedLevels.includes(currentLevelId)) {
                    completedLevels.push(currentLevelId);
                    localStorage.setItem('puz_comp_v31', JSON.stringify(completedLevels));
                }
                setTimeout(() => {
                    alert("æ­å–œé€šå…³ï¼");
                    discoverLevels();
                    showMenu();
                }, 500);
            }
        }

        function showMenu() {
            document.getElementById('game-screen').classList.remove('active');
            document.getElementById('menu-screen').classList.add('active');
        }

        discoverLevels();
    </script>
