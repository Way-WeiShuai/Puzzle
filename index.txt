        function startLevel(id) {
            currentLevel = id;
            const img = new Image();
            img.src = `${id}.jpg`; // å‡è®¾å‡ä¸ºjpgï¼Œå¯æ ¹æ®å®é™…æ ¼å¼ä¼˜åŒ–
            img.onload = () => {
                const ratio = img.width / img.height;
                cols = 3 + Math.floor((id-1)/5);
                rows = Math.round(cols / ratio);

                // è‡ªé€‚åº”æ ¸å¿ƒé€»è¾‘ï¼šè®¡ç®—å¯ç”¨åŒºåŸŸ
                const maxW = window.innerWidth * 0.92;
                const maxH = window.innerHeight * 0.65; // ä¸ºUIç•™å‡ºç©ºé—´

                // ä¼˜å…ˆé€‚é…å®½åº¦ï¼Œå¦‚æœé«˜åº¦è¶…æ ‡åˆ™æŒ‰é«˜åº¦ç¼©æ”¾
                pW = Math.floor(maxW / cols);
                pH = Math.floor((pW * cols / ratio) / rows);

                if (pH * rows > maxH) {
                    pH = Math.floor(maxH / rows);
                    pW = Math.floor((pH * rows * ratio) / cols);

                setupGame(img.src);

        function setupGame(url) {
            const board = document.getElementById('puzzle-board');
            board.innerHTML = '';
            board.style.width = `${pW * cols}px`;
            board.style.height = `${pH * rows}px`;
            document.getElementById('victory-img').src = url;
            document.getElementById('victory-img').style.opacity = '0';
            
            pieces = [];
            for(let i=0; i<rows*cols; i++) {
                pieces.push({ id: i, cur: i, bx: (i%cols)*pW, by: Math.floor(i/cols)*pH });
            }
            
            // æ‰“ä¹±
            let pool = pieces.map(p => p.id).sort(() => Math.random() - 0.5);
            pieces.forEach((p, i) => p.cur = pool[i]);

            renderPieces(url);
            document.getElementById('menu-screen').classList.remove('active');

            document.getElementById('game-info').innerText = `Level ${currentLevel} (${cols}x${rows})`;
        }

        function renderPieces(url) {
            const board = document.getElementById('puzzle-board');
            board.innerHTML = '';
            
            pieces.sort((a,b) => a.cur - b.cur).forEach(p => {
                const div = document.createElement('div');
                div.className = 'piece';
                div.style.width = `${pW}px`; div.style.height = `${pH}px`;
                div.style.backgroundImage = `url(${url})`;
                div.style.backgroundPosition = `-${p.bx}px -${p.by}px`;
                div.style.backgroundSize = `${pW*cols}px ${pH*rows}px`;
                div.dataset.id = p.id;
                
                // èåˆè§†è§‰ï¼šå¦‚æœå·¦å³/ä¸Šä¸‹ç›¸é‚»ä¸”æ­£ç¡®ï¼Œéšè—è¾¹æ¡†
                checkVisualMerge(div, p);

                div.onpointerdown = e => handleDrag(e, p.id, url);
                board.appendChild(div);
            });
        }

        function checkVisualMerge(el, p) {
            const r = Math.floor(p.cur/cols), c = p.cur%cols;
            const right = pieces.find(x => x.cur === p.cur+1);
            const down = pieces.find(x => x.cur === p.cur+cols);
            if (c < cols-1 && right?.id === p.id+1 && Math.floor(p.id/cols) === Math.floor(right.id/cols)) el.style.borderRightColor = 'transparent';
            if (r < rows-1 && down?.id === p.id+cols) el.style.borderBottomColor = 'transparent';
        }

        // --- é€»è¾‘ 2: æ‹–æ‹½ä¸èåˆ ---
        function handleDrag(e, id, url) {
            if(isMoving) return;
            const group = getGroup(id);
            const startX = e.clientX, startY = e.clientY;
            const els = group.map(gid => document.querySelector(`[data-id="${gid}"]`));
            
            els.forEach(el => { el.style.zIndex = 1000; el.style.transition = 'none'; });

            window.onpointermove = (ev) => {
                const dx = ev.clientX - startX, dy = ev.clientY - startY;
                els.forEach(el => el.style.transform = `translate(${dx}px, ${dy}px)`);
            };

            window.onpointerup = (ev) => {
                window.onpointermove = null; window.onpointerup = null;
                const dx = ev.clientX - startX, dy = ev.clientY - startY;
                const dC = Math.round(dx/pW), dR = Math.round(dy/pH);

                if (dC !== 0 || dR !== 0) {
                    let canMove = true;
                    const updates = [];
                    for(let gid of group) {
                        const p = pieces.find(x => x.id === gid);
                        const nr = Math.floor(p.cur/cols)+dR, nc = (p.cur%cols)+dC;
                        if(nr<0 || nr>=rows || nc<0 || nc>=cols) { canMove = false; break; }
                        updates.push({p, to: nr*cols+nc, from: p.cur});
                    }

                    if(canMove) {
                        const targetPos = updates.map(u => u.to);
                        const victims = pieces.filter(x => targetPos.includes(x.cur) && !group.includes(x.id));
                        const vacants = updates.map(u => u.from).filter(v => !targetPos.includes(v));
                        updates.forEach(u => u.p.cur = u.to);
                        victims.forEach((v, i) => v.cur = vacants[i]);
                    }
                }
                renderPieces(url);
                checkWin();
            };
        }

        function getGroup(sid) {
            const group = new Set([sid]), queue = [sid];
            while(queue.length > 0) {
                const id = queue.shift(), p = pieces.find(x => x.id === id);
                const r = Math.floor(p.cur/cols), c = p.cur%cols;
                const neighbors = [
                    {dr:0, dc:1,  cond: n => n.id === p.id+1 && Math.floor(p.id/cols) === Math.floor(n.id/cols)},
                    {dr:0, dc:-1, cond: n => n.id === p.id-1 && Math.floor(p.id/cols) === Math.floor(n.id/cols)},
                    {dr:1, dc:0,  cond: n => n.id === p.id+cols},
                    {dr:-1,dc:0,  cond: n => n.id === p.id-cols}
                ];
                neighbors.forEach(n => {
                    const nr = r+n.dr, nc = c+n.dc;
                    if(nr>=0 && nr<rows && nc>=0 && nc<cols) {
                        const found = pieces.find(x => x.cur === nr*cols+nc);
                        if(found && !group.has(found.id) && n.cond(found)) {
                            group.add(found.id); queue.push(found.id);
                        }
                    }
                });
            }
            return Array.from(group);
        }

        }
